拷贝构造

基础类型成员

  • "拷贝构造"1
  • 拷贝构造函数也是必要时才会产生
  • 基础数据类型，编译器可以进行一个简单的成员变量初始化，值复制，不需要合成拷贝构造函数【可以用上节课的方法进行搜索测试~】
#include <iostream>

using namespace std;

class CA {
public:
    int nCA;
};

int main() {
    // 拷贝构造函数也是必要时才会产生
    CA objCA1;
    objCA1.nCA = 11;
    CA objCA2 = objCA1; // 基础数据类型，编译器可以进行一个简单的成员变量初始化，值复制，不需要合成拷贝构造函数
    cout << objCA2.nCA << endl;

    system("pause");
    return 0;
}

内部类成员

  • CA 仅有基础类型，递归，所以也没有拷贝构造函数
#include <iostream>

using namespace std;

class CA {
public:
    int nCA;
};

class CB {
public:
    CA objCA;
    int nCB;
};

int main() {
    CB objCB1;
    objCB1.nCB = 11;
    objCB1.objCA.nCA = 12;

    CB objCB2 = objCB1;
    cout << objCB2.nCB << endl;

    system("pause");
    return 0;
}
  • 写一个显示的拷贝构造
#include <iostream>

using namespace std;

class CA {
public:
    int nCA;

public:
    CA() {};
    CA(CA & obj) {
        cout << "CA 的拷贝构造函数~" << endl;
    }
};

class CB {
public:
    CA objCA;
    int nCB;
};

int main() {
    CB objCB1;
    objCB1.nCB = 11;
    objCB1.objCA.nCA = 12;

    CB objCB2 = objCB1; // 会触发拷贝构造函数
    cout << objCB2.nCB << endl;

    system("pause");
    return 0;
}

需要返回的形参

#include <iostream>

using namespace std;

class CA {
public:
    int nCA;

public:
    CA() {
        cout << "CA 的无参构造函数~" << endl;
    }
    CA(const CA & obj) {
        nCA = obj.nCA;
        cout << "CA 的拷贝构造函数~" << endl;
    }

    ~CA() {
        cout << "CA 的析构函数~" << endl;
    }
};

CA test() {
    CA obj; // 构造函数
    return obj; // 析构临时对象 obj
}

int main() {
    CA objCA0 = test(); // 拷贝构造

    system("pause");
    return 0;
}
